创建模拟数据

df_cox <- tibble(date_s = sample(as.Date.character("2010-1-1"):as.Date.character("2010-12-31"),
    500, replace = T), date_e = sample(as.Date.character("2011-1-1"):as.Date.character("2011-12-31"),
    500, replace = T), inter = rpois(500, 30), stat = rbinom(500, 1, 0.95), age = rnorm(500,
    40, 10), sex = rbinom(500, 1, 0.6), edu = sample(1:5, 500, replace = T), dis = rbinom(500,
    1, 0.7))


df_cox %>%
    tbl_summary(by = stat) %>%
    add_p() %>%
